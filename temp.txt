let currentIndex = -1; // Current matched result index

searchInput.addEventListener('input', function() {
    const query = searchInput.value.toLowerCase();
    currentIndex = -1; // Reset current index when query changes
    searchResults.innerHTML = ''; 

    if (query.length > 0) {
        // Iterate through your data or elements and filter based on the query
        // Create a new array to hold the matching items
        const matchingItems = data.filter(item => item.title.toLowerCase().includes(query));

        if (matchingItems.length > 0) {
            matchingItems.forEach(item => {
                const resultElement = document.createElement('div');
                resultElement.textContent = item.title;
                searchResults.appendChild(resultElement);
            });
        } else {
            searchResults.innerHTML = '<div>No results found.</div>';
        }
    }
});

searchInput.addEventListener('keydown', function(event) {
    if (event.key === 'Enter') {
        event.preventDefault();
        navigateResults(1);
    } else if (event.key === 'ArrowUp') {
        event.preventDefault();
        navigateResults(-1);
    } else if (event.key === 'ArrowDown') {
        event.preventDefault();
        navigateResults(1);
    }
});

function navigateResults(direction) {
    const resultElements = searchResults.querySelectorAll('div');
    const resultCount = resultElements.length;

    if (resultCount > 0) {
        currentIndex = (currentIndex + direction + resultCount) % resultCount;
        resultElements.forEach((element, index) => {
            element.classList.toggle('active', index === currentIndex);
        });

        if (currentIndex >= 0 && currentIndex < resultCount) {
            resultElements[currentIndex].scrollIntoView({
                behavior: 'smooth',
                block: 'center',
            });
        }
    }
}

        // Function to create the pop-up menu dynamically
        function createMenu() {
            var options = [
                { label: 'Option 1', action: function() { alert('You clicked Option 1!'); } },
                { label: 'Option 2', action: function() { alert('You clicked Option 2!'); } },
                { label: 'Option 3', action: function() { alert('You clicked Option 3!'); } }
            ];

            var menuDiv = document.getElementById("popupMenu");
            menuDiv.innerHTML = ''; // Clear previous content

            for (var i = 0; i < options.length; i++) {
                var optionLink = document.createElement('a');
                optionLink.href = '#'; // Setting href to "#" to avoid page reload
                optionLink.textContent = options[i].label;
                optionLink.addEventListener('click', options[i].action); // Add click event listener
                menuDiv.appendChild(optionLink);
            }
        }

        // Function to toggle the pop-up menu visibility
        function toggleMenu() {
            var menu = document.getElementById("popupMenu");
            if (menu.style.display === "block") {
                menu.style.display = "none";
            } else {
                createMenu(); // Re-create the menu every time it's shown
                menu.style.display = "block";
            }
        }
    </script>
